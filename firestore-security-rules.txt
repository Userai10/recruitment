// Copy these rules to your Firebase Console -> Firestore Database -> Rules

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Allow users to read and write their own profile
    match /userProfiles/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
    
    // Allow authenticated users to read all profiles for duplicate checking
    match /userProfiles/{document=**} {
      allow read: if request.auth != null;
    }
    
    // Allow users to read and write their own test results
    match /testResults/{document} {
      allow read, write: if request.auth != null && request.auth.uid == resource.data.userId;
      allow create: if request.auth != null && request.auth.uid == request.resource.data.userId;
    }
    
    // Allow authenticated users to read all test results for leaderboard/admin purposes
    match /testResults/{document=**} {
      allow read: if request.auth != null;
    }
    
    // Allow users to read and write their own test status
    match /userTestStatus/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
      allow create: if request.auth != null && request.auth.uid == userId;
    }
    
    // Allow authenticated users to read all test statuses for admin purposes
    match /userTestStatus/{document=**} {
      allow read: if request.auth != null;
    }
  }
}
